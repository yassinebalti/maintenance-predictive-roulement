# 5. Prédiction de durée de vie résiduelle (RUL)
# =============================================================================
class PredicteurRUL:
    """Extrapolation linéaire de l'évolution du score d'anomalie."""
    def __init__(self):
        self.historique_scores = []   # liste de (timestamp, score)
        self.seuil_critique = 5.0     # 5 × seuil anomalie

    def ajouter_score(self, timestamp: float, score: float) -> None:
        self.historique_scores.append((timestamp, score))

    def tendance_actuelle(self) -> float:
        # pente de régression linéaire sur les 7 derniers jours
        return 0.02   # score/jour

    def estimer_rul(self) -> float:
        """Retourne le nombre de jours restants avant atteinte du seuil critique."""
        dernier_score = self.historique_scores[-1][1]
        pente = self.tendance_actuelle()
        if pente <= 0:
            return float('inf')
        jours_restants = (self.seuil_critique - dernier_score) / pente
        return max(0.0, jours_restants)

    traitement = TraitementSignal()
    extracteur = ExtractionFeatures()
    detecteur = DetecteurAnomalies()
    identifieur = IdentificateurDefaut()
    predicteur = PredicteurRUL()
    api = FastAPIApp()

    # Phase d'apprentissage (30 jours de données normales)
    print("Phase d'apprentissage du normal...")
    # (simulation de collecte et entraînement)

    # Boucle temps réel
    while True:
        # 1. Acquisition
        data = edge.acquerir_donnees()

        # 2. Traitement
        signal_filtre = traitement.filtrer_passe_bande(data["vibration"])
        features = extracteur.vecteur_complet(signal_filtre, data["rpm"])

        # 3. Détection d'anomalie
        score = detecteur.score_ensemble(features)
        if detecteur.est_anomalie(score):
            # 4. Identification du type de défaut
            signature = detecteur.autoenc.encoder(features)
            identifieur.ajouter_signature(signature)
            type_def = identifieur.type_defaut(signature)

            # 5. Mise à jour du prédicteur RUL
            predicteur.ajouter_score(time.time(), score)
            rul = predicteur.estimer_rul()

            # 6. Alerte si nécessaire
            if rul < 14:
                print(f"Alerte : défaut {type_def}, RUL = {rul:.1f} jours")

        # 7. Stockage et mise à jour dashboard via WebSocket
        api.post_nouvelles_mesures(data)

        time.sleep(2)  # période d'échantillonnage (simulée)
